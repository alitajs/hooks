import { useEffect, useMemo, Fragment } from 'react';
import { Playground } from 'docz';
import { Card, Descriptions } from 'antd';
import useWindowSize from './index';

# useWindowSize

React 状态钩子，用户动态获取 窗口 resize 后的值。

## 代码演示

### 基本示例

<Playground>
{
  () => {
    const Demo = () => {
      const windowSize = useWindowSize();

      const renderDetail = useMemo( () => {

        const {interHeight, innerWidth, outerHeight, outerWidth} = windowSize;
        return (<Fragment>

         <Descriptions bordered title="窗口resize动态变化">
           <Descriptions.Item label="innerHeight">{innerHeight}</Descriptions.Item>
           <Descriptions.Item label="innerWidth">{innerWidth}</Descriptions.Item>
           <Descriptions.Item label="outerHeight">{outerHeight}</Descriptions.Item>
           <Descriptions.Item label="outerWidth">{outerWidth}</Descriptions.Item>
          </Descriptions>
        </Fragment>)
      }, [windowSize])

      return (
        <Card>
         { windowSize && renderDetail }
        </Card>
      );
    };

    return <Demo />
  }
}
</Playground>

## API

```
const windowSize = useWindowSize({waitTime: 500});

params: { waitTime?: number}
```

## 结果

|参数|説明|类型|
|------ |--------------|---------------|
|windowSize|窗口大小数据|{innerHeight: number,innerWidth: number,outerHeight: number: number}|


## 参数
|参数|説明|类型|默认值|
|--- |------|-------|-------|
|waitTime| 节流时间 | number | 400|
